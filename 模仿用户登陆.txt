#include<stdio.h>
#include<string.h>
int main()
{
	int n=3;
	char str[20] ="0";
	char password[] = "abc123";
	while(n>0)
	{
			printf("请输入密码：\n");
		    
		    scanf("%s",&str);
			if(strcmp(password,str)==0)
			{
				printf("登陆成功\n");
				return 0;
			}
			else
			{
				if(n>1)
				{
					printf("密码错误，还有%d次登陆机会\n",n-1);
					n--;
				}
				else
				{
					printf("登陆失败\n");
					break;
				}
			}
	}
	return 0;
}
